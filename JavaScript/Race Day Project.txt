let raceNumber = Math.floor(Math.random() * 1000);
var runnerCategory = 'Adult'
var registeredEarly = false;
var runnerAge = 18;
if (runnerCategory === 'Adult' && registeredEarly){
  raceNumber += 1000;
}
if(runnerAge > 18 && registeredEarly){
  console.log('Race will start at 9:30 am')
  console.log(`Your race number is ${raceNumber}`)
}if(runnerAge > 18 && !registeredEarly){
  console.log("Late adults run at 11:00 am")
}if(runnerAge < 18){
  console.log('Youth registrants run at 12:30 pm (regardless of registration)')
}else{
  console.log('the runner to see the registration desk.')
}
